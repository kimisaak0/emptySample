	Filter = D3D11_FILTER_MIN_MAG_MIP_LINEAR;
	AddressU = CLAMP;
	AddressV = CLAMP;
	AddressW = CLAMP;
	ComparisonFunc = EQUAL;
};

struct G_OUT
{
	float4 p : SV_POSITION;
	float3 n : NO0;
	float4 c : COLOR0;
	float2 t : TEXCOORD0;

{
	float4 c : SV_TARGET;
};

{
	PS_OUTPUT vOut;
	vOut.c = g_txTextureA.Sample(sample0, vIn.t) * vIn.c;
	return vOut;
}
